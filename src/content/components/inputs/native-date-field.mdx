## Import

```js
import { NativeDateField } from "@prismane/core";
```

## Usage

```jsx
<NativeDateField name="date" />
```

### Controlled

```jsx
function Demo() {
  const [value, setValue] = useState("");

  return (
    <NativeDateField
      name="date"
      value={value}
      onChange={(e) => setValue(e.target.value)}
    />
  );
}
```

### With Icon

```jsx
<NativeDateField name="date" icon={<GearSix />} />
```

### With Addon

```jsx
<NativeDateField
  name="date"
  addons={<Field.Addon position="left">Date</Field.Addon>}
/>
```
