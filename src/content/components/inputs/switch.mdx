## Import

```js
import { Switch } from "@prismane/core";
```

## Usage

```jsx
function Demo() {
  const [value, setValue] = useState(false);

  return <Switch value={value} onChange={(e) => setValue(e.target.checked)} />;
}
```

### Switch Sizes

```jsx

  const [value, setValue] = useState(false);

  return (
    <Flex direction="column" gap={fr(2)}>
      <Switch
        size="xs"
        label="Default Field:"
        value={value}
        onChange={(e) => setValue(e.target.checked)}
      />
      <Switch
        size="sm"
        label="Default Field:"

```
