## Import

```js
import { PinField } from "@prismane/core";
```

## Usage

```jsx
function Demo() {
  const [value, setValue] = useState("");

  const handleChange = (e) => setValue(e.target.value);

  return <PinField value={value} onChange={handleChange} />;
}
```
