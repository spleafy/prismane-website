## Import

```js
import { TextField } from "@prismane/core";
```

## Usage

```jsx
function Demo() {
  const [value, setValue] = useState("");

  const handleChange = (e) => setValue(e.target.value);

  return (
    <TextField value={value} onChange={handleChange} placeholder="Enter Text" />
  );
}
```

### With Icon

```jsx
function Demo() {
  const [value, setValue] = useState("");

  return (
    <TextField
      placeholder="Enter Text"
      value={value}
      onChange={(e) => setValue(e.target.value)}
      icon={<EnvelopeSimple />}
    />
  );
}
```
