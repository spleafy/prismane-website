## Import

```js
import { Circle } from "@prismane/core";
```

## Usage

```jsx preview
<Circle size={60} bg="primary" cl="white">
  <Fire size={24} />
</Circle>
```

### Versatile Component

```jsx
<Circle
  as="a"
  href="https://www.google.com"
  target="_blank"
  size={60}
  bg="base"
>
  <Fire size={20} />
</Circle>
```
