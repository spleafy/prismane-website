## Import

```js
import { Popover } from "@prismane/core";
```

## Usage

```jsx
<Popover>
  <Popover.Control>
    <Button>Toggle Popover</Button>
  </Popover.Control>
  <Popover.Panel>
    <Text
      cl={(theme) => (theme.mode === "dark" ? ["base", 200] : ["base", 700])}
    >
      This is the content of the popover!
    </Text>
  </Popover.Panel>
</Popover>
```
