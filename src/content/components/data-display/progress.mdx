## Import

```js
import { Progress } from "@prismane/core";
```

## Usage

```jsx
<Progress value={75} label="75%" />
```

### Progress Sizes

```jsx
<Stack grow>
  <Progress value={75} label="75%" size="xs" />
  <Progress value={75} label="75%" size="sm" />
  <Progress value={75} label="75%" size="base" />
  <Progress value={75} label="75%" size="md" />
  <Progress value={75} label="75%" size="lg" />
</Stack>
```
