## Import

```js
import { useKeyboardShortcut } from "@prismane/core/hooks";
```

## Parameters

The `useKeyboardShortcut` hook accepts the following parameters:

| Name             | Type         | Description                                                                       | Default Value |
| ---------------- | ------------ | --------------------------------------------------------------------------------- | ------------- |
| `keys`           | _string[]_   | The key combination that will trigger the callback.                               | -             |
| `callback`       | _() => void_ | The callback that will be executed on a successful key combination.               | -             |
| `shouldRegister` | _boolean_    | A boolean to handle dynamic registering and unregistering of the event listeners. | _true_        |

## Usage

```jsx preview
function Demo() {
  const [registered, setRegistered] = useState(false);

  useKeyboardShortcut(["shift", "k"], () => {
    setRegistered(true);
  });

  return (
    <Text
      cl={(theme) => (theme.mode === "dark" ? ["base", 200] : ["base", 700])}
    >
      Hold: Shift + K <br />
      Registered: {registered.toString()}
    </Text>
  );
}
```

### Conditional Registering

Pass the _shouldRegister_ parameter, which is the third parameter, you can conditionally register and unregister the event listeners.

```jsx preview
function Demo() {
  const [enabled, setEnabled] = useState(false);

  const [registered, setRegistered] = useState(false);

  useKeyboardShortcut(
    ["shift", "k"],
    () => {
      setRegistered(true);
    },
    enabled
  );

  return (
    <Stack>
      <Button onClick={() => setEnabled(true)}>
        {enabled ? "Enabled" : "Enable Register"}
      </Button>
      <Text
        cl={(theme) => (theme.mode === "dark" ? ["base", 200] : ["base", 700])}
      >
        Hold: Shift + K <br />
        Registered: {registered.toString()}
      </Text>
    </Stack>
  );
}
```
