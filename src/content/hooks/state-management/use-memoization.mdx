---
title: 'useMemoization'
description: 'useMemoization hook provides a simple way to memoize expensive calculations.'
---

## Import

```js
import { useMemoization } from '@prismane/core/hooks';
```

## Return Value

The `useMemoization` hook returns the following variable:

| Name      | Type         | Description                                            |
| --------- | ------------ | ------------------------------------------------------ |
| `memoize` | _() => void_ | A function that caches the result of the passed value. |

## Usage

```jsx preview
function Demo() {
  const { memoize, clear } = useMemoization();

  const [value, setValue] = useState("");

  const [timesComputed, setTimesComputer] = useState(0);

  const expensiveFunction = memoize((pk, pv) => {
    setTimesComputer((pc: number) => pc + 1);
    return pv + 1;
  });

  const handleClick = () => {
    const result = expensiveFunction("number", 5);
    setValue(result);
  };

  return (
    <Stack>
      <Text>Value from cache: {value}</Text>
      <Text>Times computed: {timesComputed}</Text>
      <Flex gap={fr(2)} wrap="wrap">
        <Button onClick={handleClick}>Calculate</Button>
        <Button onClick={clear}>Clear Cache</Button>
      </Flex>
    </Stack>
  );
}
```
