## Import

```js
import { useCopyToClipboard } from "@prismane/core/hooks";
```

### Return Value

The `useCounter` hook returns an array containing the following properties:

| Name              | Type                                  | Description                                                |
| ----------------- | ------------------------------------- | ---------------------------------------------------------- |
| `copiedValue`     | _string_                              | The text that has been copied to the clipboard.            |
| `copyToClipboard` | _(text: string) => Promise\<boolean>_ | A function that copies the provided text to the clipboard. |

## Usage

```jsx preview
function Demo() {
  const [copiedValue, copyToClipboard] = useCopyToClipboard();

  return (
    <Stack gap={fr(4)}>
      <Flex gap={fr(2)}>
        <Button onClick={() => copyToClipboard("Prismane")}>
          Copy "Prismane"
        </Button>
        <Button onClick={() => copyToClipboard("Is the best!")}>
          Copy "Is the best!"
        </Button>
      </Flex>
      <Text>Copied value: {copiedValue || "Nothing is copied yet!"}</Text>
    </Stack>
  );
}
```
